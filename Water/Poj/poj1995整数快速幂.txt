#include <cstdio>
#include <algorithm>
#include <iostream>
using namespace std;
long long Pow(long long a,long long b,long long mod)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
        {
            b--;
            ans=(ans*a)%mod;
        }
        else
        {
            b/=2;
            a=(a*a)%mod;
        }
    }
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int mod;
        int n;
        scanf("%d",&mod);
        scanf("%d",&n);
        long long ans=0;
        int a,b;
        for(int i=1;i<=n;i++)
        {
            scanf("%d%d",&a,&b);
            ans=(ans+Pow(a,b,mod))%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
